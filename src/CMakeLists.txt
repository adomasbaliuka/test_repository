cmake_minimum_required(VERSION 3.17)

project(TestLib)


add_library(TestLib
        # Version module.
        version/version.cpp # NOTE: THIS FILE IS GENERATED BY CMAKE DURING BUILD SYSTEM GENERATION!
        
        # Additional debug output
        )

target_compile_features(TestLib PUBLIC cxx_std_17)

target_link_libraries(TestLib
        PUBLIC # propagate all header locations to tests and demo.

        )

target_include_directories(TestLib
        PUBLIC
        ${CMAKE_CURRENT_SOURCE_DIR}
        )

# Create alias with "namespace". This is the target to be used to link to the library in higher level projects.
add_library(TestLib::TestLib ALIAS TestLib)
